## ส่วนที่ 1: การเก็บรวบรวมข้อมูลเซนเซอร์ (Data Collection) (45 นาที)
ในส่วนนี้ เราจะจำลองการเก็บข้อมูลจากเซนเซอร์อัลตราโซนิก (Ultrasonic Sensor) ผ่านโปรแกรม Wokwi เพื่อแยกแยะสภาวะ "มีสิ่งกีดขวางระยะใกล้ (Near)" และ "ไม่มีสิ่งกีดขวาง (Far)"

1. เปิดโปรแกรมจำลอง Wokwi (https://wokwi.com) และสร้างโปรเจกต์ใหม่โดยเลือกบอร์ด ESP32 (หรือ Arduino)
2. เพิ่มอุปกรณ์และต่อวงจรเซนเซอร์อัลตราโซนิก (HC-SR04) เข้ากับบอร์ด (เช่น TRIG ต่อที่ขา 5, ECHO ต่อที่ขา 18)
3. เขียนโค้ดเพื่ออ่านค่าระยะทางและพิมพ์ออกทาง Serial Monitor ดังนี้:

```cpp
#define TRIG_PIN 5
#define ECHO_PIN 18

void setup() {
  Serial.begin(115200);
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
}

void loop() {
  // สร้างสัญญาณ Trigger
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  
  // อ่านค่าเวลาและคำนวณระยะทาง
  long duration = pulseIn(ECHO_PIN, HIGH);
  float distance = duration * 0.034 / 2;
  
  // พิมพ์เฉพาะค่าระยะทางเพื่อนำไปสร้าง Dataset
  Serial.println(distance);
  delay(200); // หน่วงเวลาเพื่อให้คัดลอกข้อมูลได้ง่ายขึ้น
}
```

## ส่วนที่ 2: การสร้างโมเดล Machine Learning บน Google Colab (60 นาที)
นำไฟล์ CSV ที่ได้มาใช้ฝึกสอน (Train) โมเดลปัญญาประดิษฐ์เพื่อจำแนกสภาวะจากข้อมูลระยะทาง

1. ไปที่ Google Colab (https://colab.research.google.com) และสร้างสมุดโน้ตใหม่ (New Notebook)
2. คลิกที่ไอคอนรูปโฟลเดอร์ด้านซ้ายมือ จากนั้นอัปโหลดไฟล์ `sensor_data.csv` เข้าไปในระบบ
3. คัดลอกและรันโค้ด Python ต่อไปนี้ในเซลล์โค้ด เพื่อสร้างและประเมินโมเดล K-Nearest Neighbors (k-NN)

```python
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import accuracy_score, classification_report

# 1. โหลดชุดข้อมูล (Dataset) จากไฟล์ CSV
# ระบุ header=0 เพราะเรามีชื่อคอลัมน์ที่บรรทัดแรก
df = pd.read_csv('sensor_data.csv', header=0)

# แสดงข้อมูล 5 บรรทัดแรกเพื่อตรวจสอบความถูกต้อง
print("ตัวอย่างข้อมูล:")
print(df.head())

# 2. แยกข้อมูลลักษณะเด่น (Features) และผลเฉลย (Target)
X = df[['Distance']] # ข้อมูล (Feature)
y = df['Label']      # ผลเฉลย (Target/Label)

# 3. แบ่งข้อมูลเป็นชุดฝึกสอน (Train) และชุดทดสอบ (Test)
# test_size=0.3 หมายถึงใช้ข้อมูล 30% สำหรับทดสอบ และ 70% สำหรับฝึกสอน
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# 4. สร้างและฝึกสอนโมเดล (Train Model)
# กำหนดค่าพารามิเตอร์ k=3 (พิจารณาเพื่อนบ้านที่ใกล้ที่สุด 3 จุด)
model_knn = KNeighborsClassifier(n_neighbors=3)
model_knn.fit(X_train, y_train)

# 5. ทดสอบโมเดล (Test Model) และหาความแม่นยำ (Accuracy)
predictions = model_knn.predict(X_test)
accuracy = accuracy_score(y_test, predictions)

print(f"\nความแม่นยำของโมเดล (Accuracy): {accuracy * 100:.2f}%")
print("\nรายงานผลการจำแนกประเภท (Classification Report):")
print(classification_report(y_test, predictions))
```

4. บันทึกผลลัพธ์ความแม่นยำที่ได้รันจากเซลล์แรกลงในตารางด้านล่าง
5. **ความท้าทายเพิ่มเติม:** ให้สร้างเซลล์โค้ดใหม่ และลองใช้โมเดล Decision Tree โดยใช้โค้ดตัวอย่างนี้เป็นแนวทาง:

```python
# นำเข้าโมเดล Decision Tree
from sklearn.tree import DecisionTreeClassifier

# สร้างและฝึกสอนโมเดล
model_dt = DecisionTreeClassifier()
model_dt.fit(X_train, y_train)

# นำโมเดลไปทำนายผลและหาความแม่นยำ
pred_dt = model_dt.predict(X_test)
acc_dt = accuracy_score(y_test, pred_dt)
print(f"ความแม่นยำของ Decision Tree: {acc_dt * 100:.2f}%")
```
